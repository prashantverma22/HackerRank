vector<int> compareTriplets(vector<int> a, vector<int> b) {
    vector<int> result;
    int alice = 0, bob = 0;
    for(int i=0; i<3; i++){
        if(a[i] > b[i]){
            alice += 1;
        }
        else if(a[i] < b[i]){
            bob += 1;
        }
    }
    result.push_back(alice);
    result.push_back(bob);
    return result;
}

time complexity : O(1)
space complexity : O(n), where n is the size of the result and the input arrays.
